const parts = [
    { name: "Ford Figo Headlight", model: "Figo", img: "https://via.placeholder.com/150" },
    { name: "Ford EcoSport Clutch Plate", model: "EcoSport", img: "https://via.placeholder.com/150" },
    { name: "Ford Ranger Brake Pad", model: "Ranger", img: "https://via.placeholder.com/150" },
    { name: "Ford Aspire Bumper", model: "Aspire", img: "https://via.placeholder.com/150" },
];

const partsGrid = document.getElementById('partsGrid');
const searchInput = document.getElementById('searchInput');

function renderParts(filter = '') {
    partsGrid.innerHTML = '';
    parts.filter(p => p.name.toLowerCase().includes(filter.toLowerCase()) || p.model.toLowerCase().includes(filter.toLowerCase()))
         .forEach(part => {
             const card = document.createElement('div');
             card.className = 'bg-white rounded-2xl shadow p-4';
             card.innerHTML = `
                <img src="${part.img}" alt="${part.name}" class="w-full h-40 object-cover rounded-xl mb-2">
                <h3 class="text-xl font-semibold">${part.name}</h3>
                <p class="text-gray-600">Model: ${part.model}</p>
                <a href='tel:+9779811111111' class='block mt-2 bg-blue-500 text-white text-center py-2 rounded-xl'>Call to Check Availability</a>
             `;
             partsGrid.appendChild(card);
         });
}

searchInput.addEventListener('input', e => renderParts(e.target.value));
renderParts();
</script>
